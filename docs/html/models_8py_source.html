19 def mlp(layers, units_per_layer, dropout=None):
20  """
21  Defines a classical neural network sometimes called: Multi-Layer Perceptron. It is Feedforward and fully-connected,
22  with the specified parameters. The activation function of the hidden layers is ReLU, and the activation of the
23  output is linear.
24  @param layers: Number of hidden layers (besides the input and outputs ones).
25  @param units_per_layer: Number of neurons per layer. Applies to all layers except for the last one which has only 1:
26  the predicted angle.
27  @param dropout: A value between 0 and 1 of neuron's results to discard of the training. If provided, two layers of
28  this regularization method will be added to the model. One after the input layer and one before the output layer.
29  @return model: The model of the MLP created for later usage as the predictor.
30  """
31 
32  Env.print_text(f"Creating a fully-connected feed-forward Neural Network model with {layers} layers, using "
33  f"{units_per_layer} units per layer. \nThe activation function in the hidden layers is 'ReLu' and "
34  f"the output layer has only 1 unit (the predicted angle).")
35  model = tf.keras.Sequential()
36 
37  # Shape: (time, features) => (time*features)
38  model.add(tf.keras.layers.Flatten())
39 
40  # First layer (need to specify the input size)
41  model.add(tf.keras.layers.Dense(
42  units=units_per_layer,
43  activation=tf.nn.relu))
44 
45  if dropout is not None:
46  model.add(tf.keras.layers.Dropout(dropout))
47 
48  # Other hidden layers
49  for n in range(1, layers):
50  model.add(tf.keras.layers.Dense(
51  units=units_per_layer,
52  activation=tf.nn.relu))
53 
54  if dropout is not None:
55  model.add(tf.keras.layers.Dropout(dropout))
56 
57  # Output layer
58  model.add(tf.keras.layers.Dense(
59  units=1
60  ))
61 
62  # Add back the time dimension.
63  # Shape: (outputs) => (1, outputs)
64  model.add(tf.keras.layers.Reshape([1, -1]))
65 
66  return model

69 def cnn(inputs, units_per_layer):
70  """
71  Creates a model of a Convolutional Neural Network with a Conv1D as the input layer, one dense as the only hidden
72  layer and another dense with only 1 neuron as the output layer. The first two layer has ReLU activation and the last
73  one uses a linear activation function.S
74  @param inputs: Number of inputs of the network. It is used as the kernel size with the intention that the 1D
75  convolutional layer outputs a single value throughout the specified number of filters.
76  @param units_per_layer: Specified the number of neurons of the hidden dense layer, which has to match with the
77  number of filters that will output a result in the 1D convolutional input layer.
78  @return conv_model: The model of the CNN created for later usage as the predictor.
79  """
80  Env.print_text(f"Creating a Convolutional Neural Network model with one 1D convolutional layer, using "
81  f"{units_per_layer} filters and units in the following dense layer. \nThe activation function in "
82  f"the dense layer is 'ReLU' and the output dense layer has only 1 unit (the predicted angle).")
83 
84  conv_model = tf.keras.Sequential([
85  tf.keras.layers.Conv1D(filters=units_per_layer,
86  kernel_size=(inputs,),
87  activation='relu'),
88  tf.keras.layers.Dense(units=units_per_layer, activation='relu'),
89  tf.keras.layers.Dense(units=1),
90  ])
91 
92  return conv_model

95 def lstm(units_per_layer):
96  """
97  Creates a model of a Long Short-Term Memory Neural Network as the input layer, one dense with only 1 neuron as the
98  output layer. The activation functions of the LSTM layer are the regular ones for each of it's gates.
99  @param units_per_layer: Number of inputs of the network.
100  @return lstm_model: The model of the LSTM created for later usage as the predictor.
101  """
102 
103  Env.print_text(f"Creating a Long Short-term Memory (LSTM) Neural Network using {units_per_layer} units per layer."
104  f"\nThis network only outputs the final timestamp, giving the model time to warm up its internal "
105  f"state before making a single prediction.")
106 
107  lstm_model = tf.keras.models.Sequential([
108  # Shape [batch, time, features] => [batch, time, lstm_units]
109  tf.keras.layers.LSTM(units_per_layer, return_sequences=False),
110  # Shape => [batch, time, features]
111  tf.keras.layers.Dense(units=1)
112  ])
113 
114  return lstm_model

117 def compile_and_fit(model, train, val, epochs=20, patience=2):
118  """
119  Fits the provided training data in the specified model and train's it. The validation data is used to compare the
120  performance of the model against unknown data. MSE is used as the cost function to train the model and MAE as the
121  performance evaluation metric.
122  @param model: Model to train. It can be a MLP, CNN or LSTM, among others.
123  @param train: The dataset for training the model. Must have the shape: (batch, time, features)
124  @param val: The dataset for validating the model. Must have the shape: (batch, time, features)
125  @param epochs: Number of iteration over the entire set of data (all the batches).
126  @param patience: Number of epochs to wait for improvement in the metrics. If there is no notorious improvement in
127  the performance of the validation set after the 'patience' epochs, the training will stop at this point.
128  @return history: The results of the training.
129  """
130 
131  Env.print_text(f"Compiling the input model {model.name} with {epochs} epochs. The "
132  f"loss function is the MSE and metric to evaluate improvement is the MAE.")
133 
134  early_stopping = tf.keras.callbacks.EarlyStopping(monitor='val_loss',
135  patience=patience,
136  mode='min')
137 
138  model.compile(loss=tf.losses.MeanSquaredError(), # tf.losses.mean_absolute_percentage_error()
139  metrics=[tf.metrics.MeanAbsoluteError()])
140 
141  history = model.fit(train, epochs=epochs,
142  validation_data=val,
143  callbacks=[early_stopping])
144 
145  return history
