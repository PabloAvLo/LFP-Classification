27  """
28  @details
29  <h1> Dataset Generator:</h1>
30  <h2> Setup </h2>
31  <ul>
32  <li> Initialize Environment.
33  <li> Initialize Tensorflow session.
34  <li> Set seed for Numpy and Tensorflow
35  <li> Specify run configuration parameters.
36  <li> Specify dataset generation parameters.
37  </ul>
38  <h2> Procedure </h2>
39  <ul>
40  <li> <b>Step 1:</b> Import LFP data.
41  <li> <b>Step 2:</b> Import angles data.
42  <li> <b>Step 3:</b>
43  <ul>
44  <li> Downsample LFP data to match the Angles sampling rate or
45  <li> Fill angles gaps to match the LFP sampling rate.
46  </ul>
47  <li> <b>Step 4:</b> Label data by concatenating LFPs and Angles in a single 2D-array.
48  <li> <b>Step 5:</b> Clean the labeled dataframe from -1 values, which represent the wrongly acquired
49  positional samples.
50  <li> <b>Step 6:</b> Interpolate angles data using an 'interpolation' approach.
51  <li> <b>Step 7:</b> Plotting Angles and the selected channel of the LFPs after cleaning and interpolation.
52  <li> <b>Step 8:</b>
53  <ul>
54  <li> Get preferred angle according to LFPs channel.
55  <li> Plotting the sum of LFPs per angle from 0 to 359 degrees.
56  </ul>
57  <li> <b>Step 9:</b> Get largest subset of data.
58  <li> <b>Step 10:</b> Split the subset into training, validation and testing set.
59  <li> <b>Step 11:</b> Save the dataset to a pickle file.
60  <li> <b>Step 12:</b> Convert data to windowed series.
61  <li> <b>Finish Test and Exit: </b> Save logs, captures, results and the generated dataset into a .pickle file.
62  </ul>
63  """
64 
65  # ------- SETUP -------
66  Env.init_environment(True)
67 
68  tf.keras.backend.clear_session()
69  tf.random.set_seed(51)
70  np.random.seed(51)
71 
72  # Dataset Generation Parameters
73  # The recording session 771 has a 23.81% of invalid positions, while the session 765 has only 6.98%
74  session = 771 # 771 or 765
75  interpolation = "Shortest Angle" # "linear" "quadratic" "cubic" "nearest" "Shortest Angle"
76  sync_method = "Downsample LFPs" # "Upsample Angles" "Downsample LFPs"
77 
78  # Data Properties
79  num_channels = data.EC014_41_NUMBER_OF_CHANNELS
80  rate_used = data.POSITION_DATA_SAMPLING_RATE
81  if sync_method == "Upsample Angles":
82  rate_used = data.LFP_DATAMAX_SAMPLING_RATE
83 
84  # Windowing properties
85  window_size = 4 # Recommended between 100ms to 200ms.
86  batch_size = 32
87  shuffle_buffer = 1000
88  lfp_channel = 70
89  round_angles = False
90  base_angle = 0 # Unused if round_angles = False
91  offset_between_angles = 30 # Unused if round_angles = False
92 
93  extra = {"Round angles to get discrete label": round_angles}
94  if round_angles:
95  extra.update({"Angle labels starting from": str(base_angle) + "°",
96  "until 360° in steps of": str(offset_between_angles) + "°"})
97 
98  o_pickle_file_name = f"S-{session}_C{lfp_channel}_I-{interpolation}_F-{rate_used}_W-" \
99  f"{int(window_size * 1e3 / rate_used)}ms.pickle"
100 
101  parameters_dictionary = {"Recording Session Number": str(session),
102  "Interpolation Method": interpolation.title(),
103  "Synchronization Method": sync_method,
104  "Number of Recorded Channels": str(num_channels),
105  "Sampling Rate to Use": str(rate_used) + " Hz",
106  "Window Size": window_size,
107  "Batch size (# of windows)": batch_size,
108  "LFP Signal channel to use": lfp_channel,
109  "Shuffle buffer size": shuffle_buffer}
110 
111  parameters_dictionary.update(extra)
112  Env.print_parameters(parameters_dictionary)
113 
114  # ------- STEP 1 -------
115  Env.step(f"Importing LFP data from session: {session}")
116  lfp_data = data.load_lfp_data(data.LFP[session])
117 
118  # ------- STEP 2 -------
119  Env.step(f"Importing angles data from session: {session}")
120  angles_data = data.load_angles_data(data.ANGLES[session])
121 
122  # ------- STEP 3 -------
123  interpolation_ = ""
124  if sync_method == "Downsample LFPs":
125  Env.step("Downsample LFP data to match Angles sampling rate.")
126  lfp_data = data.downsample_lfps(lfp_data, data.LFP_DATAMAX_SAMPLING_RATE, data.POSITION_DATA_SAMPLING_RATE)
127 
128  elif sync_method == "Upsample Angles":
129  Env.step("Upsample Angles data to reach a higher sampling rate")
130  angles_data = data.angles_expansion(angles_data, data.POSITION_DATA_SAMPLING_RATE,
131  data.LFP_DATAMAX_SAMPLING_RATE)
132  interpolation_ = f"Interpolados con: {interpolation}"
133 
134  # ------- STEP 4 -------
135  Env.step("Label data by concatenating LFPs and interpolated Angles in a single 2D-array.")
136  # IF round_angles: Rounding the angles to be discrete labels, starting from 'base_angle' until 360 on steps of
137  # 'offset_between_angles'. Else, the angles are not rounded to discrete labels.
138  labeled_data = data.add_labels(lfp_data, np.expand_dims(angles_data, axis=1), round_angles,
139  base_angle, offset_between_angles)
140 
141  # ------- STEP 5 -------
142  Env.step("Clean the labeled dataset from discontinuities in positional data (angles).")
143 
144  clean_datasets = data.clean_invalid_positional(labeled_data, is_padded=(sync_method == "Upsample Angles"))
145 
146  # ------- STEP 6 -------
147  Env.step(f"Interpolate angles data using a {interpolation} approach.")
148 
149  # Get all LFP channels, excluding the angles.
150  clean_interpolated_data = [s[:, :-1] for s in clean_datasets]
151 
152  # Interpolate angles and add them to the dataset
153  clean_interpolated_angles = [data.interpolate_angles(s[:, -1], interpolation) for s in clean_datasets]
154 
155  for i in range(len(clean_interpolated_data)):
156  clean_interpolated_data[i] = \
157  np.concatenate((clean_interpolated_data[i], np.expand_dims(clean_interpolated_angles[i], axis=1)), axis=1)
158 
159  # ------- STEP 7 -------
160  Env.step("Plotting Angles and one channel of the LFPs after cleaning and interpolation.")
161 
162  Env.print_text(f"Plotting Angles data [°] at {rate_used}Hz")
163  figname = f"{session}_Angles_degrees_{int(rate_used)}Hz"
164  plt.figure(figname)
165  plt.plot(clean_interpolated_data[0][0:round(3*60*rate_used), -1], "xr")
166  plt.title(f"3 minutos de Ángulos [°] limpios.\n {interpolation_}. Sesión: {session} a {rate_used}Hz")
167  ui.store_figure(figname, show=Env.debug)
168 
169  Env.print_text(f"Plotting LFPs Channel {lfp_channel} at {rate_used}Hz")
170 
171  figname = f"{session}_LFP_c{lfp_channel}_{int(rate_used)}Hz"
172  plt.figure(figname)
173  plt.plot(clean_interpolated_data[4][:, lfp_channel], "x-r")
174  plt.title(f"LFPs del Canal {lfp_channel}. Sesión: {session} a {rate_used}Hz")
175  ui.store_figure(figname, show=Env.debug)
176 
177  Env.print_text(f"Plot clean LFP data from the chosen channel and interpolated angles data at {rate_used}Hz. [°]")
178 
179  figname = f"{session}_LFP_C{lfp_channel}_clean_and_angles_{interpolation}_{rate_used}Hz"
180  fig, axs = plt.subplots(2, 1, num=figname, constrained_layout=True)
181  fig.suptitle(f"5 minutos limpios de la sesión: {session} a {rate_used}Hz.")
182  axs[0].plot(clean_interpolated_data[0][0:round(5*60*rate_used), lfp_channel], "xr")
183  axs[0].set_title(f"Señal LFP del Canal {lfp_channel}.")
184  axs[1].plot(clean_interpolated_data[0][0:round(5*60*rate_used), -1], "xb")
185  axs[1].set_title(f"Ángulos [°]. {interpolation_}.")
186  ui.store_figure(figname, show=Env.debug)
187 
188 
189  # ------- STEP 8 -------
190  Env.step("Get Preferred angle.")
191 
192  angles = [0] * 361
193 
194  for subset in clean_interpolated_data:
195  for index in range(len(subset)):
196  angles[int(round(subset[index, -1]))] += abs(subset[index, lfp_channel])
197 
198  angles[0] += angles[360]
199  angles = angles[:-1]
200  preferred_angle = angles.index(max(angles))
201 
202  Env.print_text(f" Preferred angle according to LFP Channel {lfp_channel}: {preferred_angle}°")
203 
204  Env.print_text(f"Plotting Preferred angle according to LFP Channel {lfp_channel}")
205  figname = f"{session}_preferred_angle_LFP_c{lfp_channel}_{int(rate_used)}Hz"
206  plt.figure(figname)
207  plt.plot(range(360), angles, "x-r")
208  plt.title(f"Suma de LFPs del Canal {lfp_channel} por ángulo. Sesión: {session}.")
209  plt.xlabel(f"Ángulo en grados")
210  ui.store_figure(figname, show=Env.debug)

212  # ------- STEP 9 -------
213  Env.step("Get largest subset of data.")
214 
215  max_length = 0
216  max_subset_index = 0
217  for index, subset in enumerate(clean_interpolated_data):
218  if max_length < len(subset):
219  max_length = len(subset)
220  max_subset_index = index
221 
222  largest_subset = clean_interpolated_data[max_subset_index]
223  Env.print_text(f"Shape of the largest subset of data : [{len(largest_subset)},{len(largest_subset[0])}]")
224 
225  # ------- STEP 10 -------
226  Env.step()
227

229  Env.print_text(f"Largest subset: {max_subset_index} = {n} samples")
230  train_array = largest_subset[0:int(n * 0.7), :]
231  valid_array = largest_subset[int(n * 0.7):int(n * 0.9), :]
232  test_array = largest_subset[int(n * 0.9):, :]
233 
234  Env.print_text(f"Training data shape: [{len(train_array)},{len(train_array[0])}]")

247  train_data = data.channels_to_windows(train_array, lfp_channel, window_size, batch_size, shuffle_buffer)

251  # The shape should be (batch, time, features) to be compatible with what tensorflow expects as default.
252  for example_inputs, example_labels in train_data.take(1):

254  Env.print_text(f'Labels shape (batch, time, labels): {example_labels.shape}')
