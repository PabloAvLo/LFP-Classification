59 def load_angles_data(file=ANGLES[771], degrees=True):
60  """
61  Loads the animal position data from a .whl file which contain 2 (x, y) pairs, one for each LED. If any position
62  value equals '-1' then it's replaced with 'NaN' instead.
63  @param file: Path to the file containing the animal LED's position information
64  @param degrees: If this flag is set, then the angles are returned in degrees from [0, 360[, or radians otherwise.
65  @return angles: Array with the angles in radians extracted from the positions. The angles are given as
66  float16 values calculated as arctan(y2 - y1 /x2 - x1). Unless the denominator is 0, in that case '0' is returned
67  for that element.
68  """
69  Env.print_text("Loading the animal position data from: " + os.path.basename(file) + ".")
70 
71  positions_file = open(file, "rb")
72  positions = np.genfromtxt(fname=positions_file, dtype=np.float16, delimiter='\t')
73  positions_file.close()
74 
75  positions[positions == -1] = np.NAN
76  #TODO: of x_2 - x_1 = 0, angles should be NaN.
77 
78  angles = np.arctan2((np.subtract(positions[:, 3], positions[:, 1])),
79  (np.subtract(positions[:, 2], positions[:, 0])))
80 
81  if degrees:
82  angles = np.degrees(angles)
83  angles += 180
84 
85  angles[np.isnan(angles)] = -1
86  invalid = np.count_nonzero(angles == -1)
87 
88  Env.print_text("Head angle data shape: " + str(np.shape(angles)))
89  Env.print_text(f"Head angle invalid data: {invalid:d} ({invalid/len(angles) * 100:.2f}%)")
90 
91  return angles

94 def downsample_lfps(lfp_data, orig_rate, new_rate):
95  """
96  Downsample the LFP signal data after applying an anti-aliasing filter.
97  An order 8 Chebyshev type I filter is used. Usually the LFP signals are acquired at a higher sampling rate
98  than the position data.
99  @note: This method assumes that the reason of frequencies is 32 to compute the decimation.
100  @param lfp_data: Matrix [n x numChannels] with the LFP signals
101  @param orig_rate: Sampling rate originally used to acquire the data.
102  @param new_rate: New sampling rate of the data.
103  @return resampled_data: Original data downsampled to the new rate.
104  """
105  Env.print_text("Downsampling LFP data to match the new sampling rate: " + str(new_rate)
106  + "Hz. Original was: " + str(orig_rate) + "Hz.")
107 
108  resampled_data = []
109  for channel_i in np.transpose(lfp_data):
110  channel_i = scipy.signal.decimate(channel_i, 8)
111  channel_i = scipy.signal.decimate(channel_i, 4)
112  resampled_data.append(channel_i)
113 
114  resampled_data = np.transpose(resampled_data)
115  Env.print_text("LFP data decimated shape: " + str(np.shape(resampled_data)))
116 
117  return resampled_data

120 def angles_expansion(angles_data, orig_rate, new_rate):
121  """
122  Fill angular data with 'NaN' values to match an expected sampling rate. Usually the position data is acquired at a
123  lower sampling rate than the LFP signals.
124  @details Assuming that the acquisition of data started and stopped at the same time, then no data has to be added
125  after the last sample.
126  @param angles_data: Array with the angles data extracted from the animal positions.
127  @param orig_rate: Sampling rate originally used to acquire the data.
128  @param new_rate: New sampling rate of the data. The gaps are filled with 'NaN'.
129  @return upsampled_data: Original data filled with 'NaN' to match the new sampling rate.
130  """
131  Env.print_text("Expanding angles data with 'NaN' values to match the new sampling rate: " + str(new_rate) + "Hz. "
132  + "Original was: " + str(orig_rate) + "Hz.")
133 
134  expansion_factor = round(new_rate/orig_rate)
135  padding = np.full((len(angles_data), expansion_factor - 1), np.NaN)
136  upsampled_data = np.concatenate((np.transpose(np.array([angles_data])), padding), axis=1)
137  upsampled_data = upsampled_data[:-1, :]
138  upsampled_data = upsampled_data.flatten()
139 
140  Env.print_text("Angles data upsampled shape: " + str(np.shape(upsampled_data)))
141 
142  return upsampled_data

145 def shortest_angle_interpolation(start, end, amount):
146  """
147  This interpolation method considers the 'start' and 'end' as angles in a circumference where the objective is to
148  find the smallest arch between two angles.
149  param start: Start angle with values in the range of [0 to 360[
150  param end: Final angle with values in the range of [0 to 360[
151  param amount: Value between [0, 1] which determines how close the interpolated angle will be placed from the Start
152  angle (0) or from the Final angle (1), being 0.5 the middle.
153  return interpolated_angle: Interpolated angle between 'start' and 'end'.
154  """
155  shortest_angle = ((end-start) + 180) % 360 - 180
156  interpolated_angle = (start + shortest_angle * amount) % 360
157  return interpolated_angle

160 def vectorized_sai(angles_data):
161  """
162  Replace 'NaN' values between valid values (interpolation) in angular data with an interpolated value using
163  the shortest angle interpolation.
164  param angles_data: Array with the angles data extracted from the animal positions.
165  return interpolated_angles: Array with the angles data interpolated using the Shortest Angle Interpolation
166  """
167  start_angle = np.nan
168  no_nans = 0
169  first_nan_index = 0
170  interpolated_angles = angles_data
171 
172  for i in range(1, len(interpolated_angles)):
173  # If a valid value followed by NaN: this is the first NaN, start counting
174  if np.isnan(interpolated_angles[i]) and not np.isnan(interpolated_angles[i-1]):
175  start_angle = interpolated_angles[i-1]
176  no_nans = 1
177  first_nan_index = i
178 
179  # If a NaN followed by another NaN: Increment counter 1+.
180  elif np.isnan(interpolated_angles[i]) and np.isnan(interpolated_angles[i - 1]):
181  no_nans += 1
182 
183  # If a NaN followed by a valid value: This is the last NaN, interpolate.
184  elif not np.isnan(interpolated_angles[i]) and np.isnan(interpolated_angles[i - 1]):
185  if no_nans > 0 and not np.isnan(start_angle):
186  amount = 0
187  end_angle = interpolated_angles[i]
188  for j in range(first_nan_index, first_nan_index + no_nans):
189  amount += 1/(no_nans + 1)
190  interpolated_angles[j] = shortest_angle_interpolation(start_angle, end_angle, amount)
191  start_angle = np.nan
192 
193  return interpolated_angles

196 def interpolate_angles(angles_data, method="Shortest"):
197  """
198  Replace 'NaN' values in angular data with an interpolated value using a given method.
199  @param angles_data: Array with the angles data extracted from the animal positions.
200  @param method: Interpolation method to fill the gaps in the data. The optional methods available are the supported
201  by pandas.DataFrame.interpolate function, which are: 'linear', 'quadratic', 'cubic', 'polynomial', among others.
202  @return interpolated_angles: Array with the angles data interpolated using the given method.
203  """
204  Env.print_text("Interpolate angles data using " + method + " method.")
205 
206  if method == "Shortest":
207  interpolated_angles = vectorized_sai(angles_data)
208 
209  else:
210  angles_series = pd.Series(angles_data)
211  interpolated_angles = angles_series.interpolate(method)
212  interpolated_angles = interpolated_angles.to_numpy()
213 
214  return interpolated_angles

217 def add_labels(lfps, angles, round_labels, start=0, offset=30):
218  """
219  Add an additional column to the LFP signals matrix with the angular data used as the labels.
220  @param lfps: Matrix [n x numChannels] with the LFP signals used as the preliminary features of the data.
221  @param angles: Array with the angles data extracted from the positions used as the labels of the data.
222  @param round_labels: Boolean, if true the labels are rounded to angles multiples of 'offset' starting from 'start'
223  @param start: Angle in [0째, 360째[ used as first label.
224  @param offset: Offset in [1째, 360째[ between labels starting from 'start' angle.
225  @return labeled_data: Matrix with the labeled data [n x lfps[numChannels], angles].
226  """
227 
228  len_lfps = len(lfps)
229  len_angles = len(angles)
230 
231  if len_lfps > len_angles:
232  lfps = lfps[0: len_angles]
233  elif len_lfps < len_angles:
234  angles = angles[0: len_lfps]
235 
236  Env.print_text("Adding labels to the data by concatenating the [" + str(len(lfps)) + " x " + str(len(lfps[0])) +
237  "] LFP data matrix with the [" + str(len(angles)) + "] Angles vector.")
238  Env.print_text("Rounding Labels = " + str(round_labels))
239 
240  if round_labels:
241  if (0 <= start < 360) and (1 <= offset < 360):
242  labels = np.arange(start, 360, offset)
243 
244  for i in range(0, len(angles)):
245  if not np.isnan(angles[i]):
246  minval = np.inf
247  label = start
248  for tag in labels:
249  diff = abs(angles[i] - tag)
250  if diff < minval:
251  minval = diff
252  label = tag
253  angles[i] = label
254 
255  labeled_data = np.concatenate((lfps, angles), axis=1)
256 
257  return labeled_data

260 def clean_invalid_positional(labeled_dataset, is_padded=True):
261  """
262  Clean the data rows which have '-1' values as labels (angles) from the the data and their LFPs associated in each
263  channel. Plus the following 31 rows with NaN as angle value in case of padded data.
264  @details The positional data taken from the LEDs placed in the rat have discontinuities where one or both LEDs
265  are lost, making them invalid.
266  Hence '-1' values are used instead to denote invalid position data and are meant to be removed from the data since
267  they are not representative labels.
268  @param labeled_dataset: Matrix [n x (numChannels +1)] with the LFP signals used as the preliminary features of the
269  data and the angles data extracted from the positions used as the labels of the data.
270  @param is_padded: If True, manage the input labeled dataset as a padded array of angles, or a downsampled LFP set
271  otherwise.
272  @return clean_dataset: Input data without invalid positional values.
273  """
274  if is_padded:
275  # Get the indexes of all invalid values (-1) plus the 31 following padded values (NaN).
276  # and save that range as an element of 'invalid_indexes' array.
277  invalid_indexes = [np.arange(i, i + 32).tolist() for i, v in enumerate(labeled_dataset) if v[-1] == -1]
278  Env.print_text(f"Amount of invalid indexes: {len(invalid_indexes)}")
279 
280  # Merge all sublists as a single consecutive array of invalid indexes.
281  invalid_indexes = [item for sublist in invalid_indexes for item in sublist]
282  Env.print_text(f"Amount of invalid indexes + associated expanded indexes: {len(invalid_indexes)}")
283  Env.print_text(f"Number of valid samples: {len(labeled_dataset) - len(invalid_indexes)} (including the 31 NaN "
284  f"values padded for the last real positional data.)")
285  else:
286  # Get the indexes of all invalid values (-1) and save that range as an element of 'invalid_indexes' array.
287  invalid_indexes = [i for i, v in enumerate(labeled_dataset) if v[-1] == -1]
288  Env.print_text(f"Amount of invalid indexes: {len(invalid_indexes)}")
289  Env.print_text(f"Number of valid samples: {len(labeled_dataset) - len(invalid_indexes)}")
290 
291  # Get the indexes where the discontinuities start and end.
292  discontinuities_starts = [invalid_indexes[0]]
293  discontinuities_ends = []
294  for i in range(1, len(invalid_indexes)):
295  if invalid_indexes[i] - 1 != invalid_indexes[i-1]:
296  discontinuities_starts.append(invalid_indexes[i])
297  discontinuities_ends.append(invalid_indexes[i-1])
298  discontinuities_ends.append(invalid_indexes[-1])
299 
300  # Stores sub-arrays of valid indexes:
301  clean_datasets = []
302 
303  # From the first item to the start of the first discontinuity
304  if discontinuities_starts[0] != 0:
305  clean_datasets.append(labeled_dataset[0:discontinuities_starts[0], :])
306 
307  # From the end+1 of the ith discontinuity to the start (not included) of the ith + 1 discontinuity
308  for i in range(len(discontinuities_starts)-1):
309  clean_datasets.append(labeled_dataset[discontinuities_ends[i]+1:discontinuities_starts[i+1], :])
310 
311  # From the end of the last discontinuity to the last item.
312  if discontinuities_ends[-1] != len(labeled_dataset)-1:
313  clean_datasets.append(labeled_dataset[discontinuities_ends[-1] + 1:-1, :])
314 
315  # Delete subsets with only 1 valid sample (i.e its length is <=32)
316  # From the resulting subsets, delete the last 31 rows with NaN angles
317  clean_datasets = [v[:-31, :] for v in clean_datasets if len(v) > 32]
318 
319  Env.print_text(f"Total number of valid subsets: {len(clean_datasets)}")
320  for s in range(0, len(clean_datasets)):
321  Env.print_text(f"Total number of samples and channels + angles in subsets {s}: {clean_datasets[s].shape}")
322 
323  return clean_datasets

348 def channels_to_windows(series, channel, window_size, batch_size, shuffle_buffer=None, offset=1):
349  """
350  Receives a numpy array containing the time series of LFP signals of n channels and returns the same data, separated
351  in windows.
352  @param series: Numpy Array with the LFP data of the n channels.
353  @param channel: Channel to use.
354  @param window_size: Size of the windows in which the data are being split.
355  @param batch_size: Number of pairs data-labels to group as a batch
356  @param shuffle_buffer: Number of windows to shuffle at the same time.
357  @param offset: Number of samples to shift between windows.
358  @return windowed_ds: LFP data of the selected channel separated in windows.
359  """
360 
361  labels = np.expand_dims(series[window_size::, -1], 1)
362  data = series[:, channel]
363 
364  # Get the average angle for each window.
365  #labels_ds = average_angles(labels, window_size)
366 
367  labels_ds = tf.data.Dataset.from_tensor_slices(np.expand_dims(labels, 1))
368 
369  # Creates a dataset from the input
370  windowed_data = tf.data.Dataset.from_tensor_slices(np.expand_dims(data, 1))
371 
372  # Split the data set in windows shifting each window by 1 and forcing them to the same size (window_size + 1)
373  windowed_data = windowed_data.window(window_size, shift=offset, drop_remainder=True)
374 
375  # Make each window a numpy array row.
376  windowed_data = windowed_data.flat_map(lambda window: window.batch(window_size))
377 
378  # Add labels to the data
379  windowed_ds = tf.data.Dataset.zip((windowed_data, labels_ds))
380 
381  if Env.debug:
382  for x, y in windowed_ds.take(1):
383  print("x = ", x.numpy())
384  print("y = ", y.numpy())
385 
386  if shuffle_buffer != None:
387  # Shuffle the data in groups of shuffle_buffer to accelerate. Instead of shuffle it all at once.
388  windowed_ds = windowed_ds.shuffle(shuffle_buffer)
389 
390  # Batch the data into sets of 'batch_size'.
391  windowed_ds = windowed_ds.batch(batch_size).prefetch(1)
392 
393  return windowed_ds
